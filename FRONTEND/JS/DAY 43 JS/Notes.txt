Math.floor() :-

    Convert The Given Value In Downward Direction
    Means :
        if -> Math.floor(4.9) -> Output -> 4
        if -> Math.floor(4.1) -> Output -> 4

Math.round() :-

    Rounding The Value Nearest
    Means:
        if -> Math.round(4.49) -> Output -> 4
        if -> Math.round(4.55) -> Output -> 5

Math.ceil() :-

    Convert The Given Value In Upward Direction
    Means:
        if -> Math.ceil(4.1) -> Output -> 5
        if -> Math.ceil(4.9) -> Output -> 5
